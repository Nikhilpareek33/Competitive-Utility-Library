const int N=1e5;
const int k=17; // for upto 10^7 k = 25 is sufficient
ll sparse[N+1][k];

void fun(vll &a,ll &n)
{
	fr(i,n)
	{
		sparse[i][0]=a[i];
		
	}

	fr1(j,17)
	{
		for(ll i=0;i+(1<<j)<=n;i++)
		{
			sparse[i][j]=min(sparse[i][j-1],sparse[i+(1<<(j-1))][j-1]); // [i,i+2j−1−1]  and [i+2j−1,i+2j−1]

		}

	}
	
}

ll ans(ll l,ll r)
{
	ll temp=floor(log2(r-l+1)); // alt (int)log2(r-l+1);

	return min(sparse[l][temp],sparse[r-(1<<temp)+1][temp]);
}
